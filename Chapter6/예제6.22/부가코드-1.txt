
/* ================= 6.6.2 System.Threading.Monitor ================= */

using System;
using System.Threading;

class Program
{
    static void Main(string[] args)
    {
        for (int i = 0; i < 10; i++)
        {
            Thread t = new Thread(threadFunc);
            t.Start(i);
        }
    }

    static void threadFunc(object value)
    {
        Console.WriteLine(value + "번째 스레드");
    }
}
