

/* ================= 6.3.1 System.BitConverter ================= */

using System;

class Program
{
    static void Main(string[] args)
    {
        byte[] buf = new byte[4];
        buf[0] = 0x4c;
        buf[1] = 0x36;
        buf[2] = 0x19;
        buf[3] = 0x00;

        int result = BitConverter.ToInt32(buf, 0);
        Console.WriteLine(result); // 출력 결과: 1652300
    }
}
