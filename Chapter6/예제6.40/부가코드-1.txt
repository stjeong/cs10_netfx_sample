class Program

    {

        IPAddress ipAddr = Dns.GetHostEntry("www.naver.com").AddressList[0];
        EndPoint serverEP = new IPEndPoint(ipAddr, 80);
        socket.Connect(serverEP);

        // …… HTTP 프로토콜 통신 ……

        socket.Close();
    }
}
